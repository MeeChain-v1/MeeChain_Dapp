const OnboardingController = {
  init() {
    this.bindEvents();
    this.showStep(onboardingState.currentStep);
    this.updateProgress();
  },

  bindEvents() {
    document.querySelector('#start-button')?.addEventListener('click', () => this.start());
    document.querySelector('#next-button')?.addEventListener('click', () => this.next());
    document.querySelector('#prev-button')?.addEventListener('click', () => this.prev());
  },

  start() {
    onboardingState.currentStep = 1;
    this.showStep(1);
    document.getElementById('onboarding-modal').style.display = 'flex';
    document.getElementById('main-content').classList.add('hidden');
  },

  next() {
    if (onboardingState.currentStep < onboardingState.totalSteps) {
      this.hideStep(onboardingState.currentStep);
      onboardingState.currentStep++;
      this.showStep(onboardingState.currentStep);
      this.updateProgress();
    }
  },

  prev() {
    if (onboardingState.currentStep > 1) {
      this.hideStep(onboardingState.currentStep);
      onboardingState.currentStep--;
      this.showStep(onboardingState.currentStep);
      this.updateProgress();
    }
  },

  showStep(step) {
    document.querySelector([data-step="${step}"])?.classList.add('active');
  },

  hideStep(step) {
    document.querySelector([data-step="${step}"])?.classList.remove('active');
  },

  updateProgress() {
    const percent = ((onboardingState.currentStep - 1) / (onboardingState.totalSteps - 1)) * 100;
    document.getElementById('onboarding-progress').style.width = ${percent}%;
    document.getElementById('current-step').textContent = onboardingState.currentStep;
  },

  complete() {
    localStorage.setItem('meechainonboardingcomplete', 'true');
    document.getElementById('onboarding-modal').style.display = 'none';
    document.getElementById('main-content').classList.remove('hidden');
    showToast('üéâ ‡∏¢‡∏¥‡∏ô‡∏î‡∏µ‡∏ï‡πâ‡∏≠‡∏ô‡∏£‡∏±‡∏ö‡∏™‡∏π‡πà MeeChain!');
  }
};
